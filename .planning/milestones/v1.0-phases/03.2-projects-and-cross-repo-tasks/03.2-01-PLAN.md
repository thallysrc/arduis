```python
"""GTK-free project/member-repo discovery (D-05/D-07). Imports NO gi."""
from __future__ import annotations
import os

def detect_member_repos(root: str) -> list[str]:
    """Direct subdirs of root that contain a .git entry (dir OR file).
    The root's own .git is NOT a member (D-05). No tree-walk-up (D-07).
    Returns sorted names; [] on error or when none (caller treats [] as
    degenerate 1-repo project)."""
    members: list[str] = []
    try:
        with os.scandir(root) as it:
            for e in it:
                if e.is_dir(follow_symlinks=False) and os.path.exists(os.path.join(e.path, ".git")):
                    members.append(e.name)
    except OSError:
        return []
    return sorted(members)
```

```python
"""GTK-free task-folder layout builders (D-08/D-09). Imports NO gi.

Threats:
- T-02-02 (path traversal): the task-dir leaf is sanitize_branch_for_dir(branch);
  task_dir_for derives from dirname(root), never from raw branch input.
- V12 (file & resources): symlink_plan only mirrors top-level root entries to
  positions INSIDE task_dir; it never resolves user input into a symlink target.
"""
from __future__ import annotations
import os
from arduis.worktree import sanitize_branch_for_dir

def task_dir_for(root: str, branch: str) -> str:
    """Grouped sibling: <parent-of-root>/<root_base>-tasks/<sanitized-branch> (D-08)."""
    root = root.rstrip("/")
    parent, base = os.path.dirname(root), os.path.basename(root)
    return os.path.join(parent, f"{base}-tasks", sanitize_branch_for_dir(branch))

def repo_worktree_dir(task_dir: str, repo_name: str) -> str:
    """A chosen repo's worktree dir inside the task folder (D-08 — repo dir names kept)."""
    return os.path.join(task_dir, repo_name)

def symlink_plan(root: str, task_dir: str, chosen_repos: set[str]) -> list[tuple[str, str]]:
    """(src_abs_in_root, dst_in_task_dir) for every top-level root entry EXCEPT
    the chosen repos (real worktrees) and the meta-repo .git (D-05/D-09).
    Pure: lists root entries, returns pairs; the caller does the os.symlink with
    a relative target so the task folder stays relocatable (Pattern 3, A1)."""
    plan: list[tuple[str, str]] = []
    for name in sorted(os.listdir(root)):
        if name in chosen_repos:
            continue            # real worktree, materialized separately (D-09)
        if name == ".git":
            continue            # meta-repo .git is NOT mirrored (D-05)
        plan.append((os.path.join(root, name), os.path.join(task_dir, name)))
    return plan
```
