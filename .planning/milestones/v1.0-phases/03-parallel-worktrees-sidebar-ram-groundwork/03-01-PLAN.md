layout.py (Plan 03-02 implements):
```python
# Pure binary tree. LeafNode holds a session_id; SplitNode holds orientation + two children.
class LeafNode:           # LeafNode(session_id: str | None)
    session_id: str | None
class SplitNode:          # SplitNode(orientation: str, start, end)  orientation in {"h","v"}
    orientation: str
    start: "LeafNode | SplitNode"
    end:   "LeafNode | SplitNode"

class LayoutModel:
    root: "LeafNode | SplitNode | None"
    focused_id: str | None
    def split(self, session_id: str, new_session_id: str, orientation: str = "h") -> None
    def close_leaf(self, session_id: str) -> None        # collapses degenerate parent
    def visible_ids(self) -> list[str]                   # session_ids that are leaves
    def is_visible(self, session_id: str) -> bool
    def set_leaf_session(self, target_id: str, new_id: str) -> None  # focus-or-swap target
    def zoom(self, session_id: str) -> None              # save tree, show only this leaf
    def unzoom(self) -> None                             # restore saved tree
    def is_zoomed(self) -> bool
    def preset(self, kind: str, session_ids: list[str]) -> None  # kind in {"grid2x2","columns"}
    def touch(self, session_id: str) -> None             # record MRU focus order
    def mru_order(self) -> list[str]                     # most-recently-focused first
def resolve_selection(model: "LayoutModel", session_id: str) -> tuple[str, str]
    # returns ("focus", session_id) if visible else ("swap", focused_id)  -- D-06, pure
```
