```python
def load_agent_config(path: str) -> AgentConfig:
    try:
        with open(path, "rb") as fh:
            data = tomllib.load(fh)
    except (OSError, tomllib.TOMLDecodeError):
        return AgentConfig()
    section = data.get("agent")
    if not isinstance(section, dict):
        return AgentConfig()
    cmd = section.get("command")
    if not isinstance(cmd, str) or not cmd.strip():
        return AgentConfig()
    return AgentConfig(command=cmd.strip())
```

<plan_decisions>
- **`.arduis.toml` schema (D-02):** ONLY `[setup].commands` (an ordered list of shell command
  strings) is read this phase. Ignore unknown sections/keys SILENTLY (forward-compat: Phase 7
  adds `[containers]` to the same file — do NOT validate-and-reject unknown keys).
- **`RepoSetup` dataclass:** `@dataclass class RepoSetup: commands: list[str] = field(default_factory=list)`.
  `commands == []` is the no-op default and the "no gate" signal.
- **`load_repo_setup(repo_dir)` tolerance (D-01, Pitfall 7):** open `<repo_dir>/.arduis.toml`;
  on `OSError`/`TOMLDecodeError` -> `RepoSetup()`. `data.get("setup")` not a dict -> `RepoSetup()`.
  `section.get("commands")` not a list -> `RepoSetup()`. Then keep only non-empty stripped
  strings, in order: `[c.strip() for c in raw if isinstance(c, str) and c.strip()]`.
- **`setup_feed_bytes(worktree_dir, commands)` (D-04/D-05, Pitfall 1/5/6):** the directory
  guard is `cd <shlex.quote(worktree_dir)> &&` followed by a newline, then each command on its
  OWN line, then a trailing `\n`; `.encode("utf-8")` (bytes at the 0.76 floor). The command
  LIST is newline-joined — NEVER `&&`-chained (one failure must not hide the rest from view).
  `cd` uses `&&` only so a failed cd does not run commands in the wrong dir. The commands
  themselves are the user's RAW shell lines — do NOT `shlex`-wrap them (they intentionally
  contain `&&`, `$VAR`, `cp a b`). Only the `cd` TARGET is `shlex.quote`d.
  Exact shape for `worktree_dir="/t/wt"`, `commands=["npm install","cp .env.example .env"]`:
  `b"cd '/t/wt' &&\nnpm install\ncp .env.example .env\n"`.
  An empty `commands` list -> caller never calls this (guarded by `commands == []`); but if
  called with `[]`, return just the `cd` guard line `b"cd '<dir>' &&\n"` is WRONG (trailing
  `&&` with nothing) — instead return `b""` for an empty list (defensive; tested).
- **`setup_hash(commands)` (D-07):** `hashlib.sha256("\n".join(commands).encode("utf-8")).hexdigest()`.
  Order is semantically meaningful (reorder MUST change the hash). Hash the RAW commands list
  (NOT the cd-guarded feed) so the trust key is about the repo's authored intent, independent
  of where the worktree lands.
- **Trust identity (D-09):** the caller (Plan 02) passes `os.path.realpath(<project_root>/<repo_name>)`
  — the repo SOURCE dir. `trust.py` treats `repo_id` as an opaque string key; it does NOT
  compute realpath itself (keeps it pure/testable and GTK-free).
- **Trust list (D-10):** `~/.config/arduis/trusted_setups.toml`, a single `[trusted]` table
  `{ "<repo_realpath>" = "<sha256hex>" }`. `load_trusted(path)` is tolerant exactly like
  `load_agent_config` (missing/garbage/`trusted` not a dict -> `{}`; drop non-str values) —
  fail-closed: an unreadable list re-prompts EVERYTHING. `record_trust(path, repo_id, hash)`
  read-merges (preserve existing entries), sets `trusted[repo_id] = hash`, re-serializes the
  WHOLE `[trusted]` table, writes atomically (tmp+os.replace, makedirs parent, swallow OSError —
  the appconfig idiom). `is_trusted(path, repo_id, hash)` -> `load_trusted(path).get(repo_id) == hash`.
- **Local TOML key serialization:** repo paths contain `/`/`.`/`-`, so each must be a QUOTED
  string key. Build lines as `f'"{_esc(k)}" = "{_esc(v)}"'` where `_esc` does
  `.replace("\\", "\\\\").replace('"', '\\"')` (mirror appconfig._fmt_scalar's escaping). Emit
  `"[trusted]\n" + "\n".join(lines) + "\n"`. Round-trip it through `tomllib` in a test to prove
  it parses back.
- **GTK-free (the discipline):** neither file imports `gi`. Tests assert
  `"import gi" not in open(<file>).read()` and `"from gi" not in ...`.
</plan_decisions>

```python
from dataclasses import dataclass, field
import os, shlex, tomllib

@dataclass
class RepoSetup:
    commands: list[str] = field(default_factory=list)

def load_repo_setup(repo_dir: str) -> RepoSetup:
    path = os.path.join(repo_dir, ".arduis.toml")
    try:
        with open(path, "rb") as fh:
            data = tomllib.load(fh)
    except (OSError, tomllib.TOMLDecodeError):
        return RepoSetup()
    section = data.get("setup")
    if not isinstance(section, dict):
        return RepoSetup()
    raw = section.get("commands")
    if not isinstance(raw, list):
        return RepoSetup()
    cmds = [c.strip() for c in raw if isinstance(c, str) and c.strip()]
    return RepoSetup(commands=cmds)

def setup_feed_bytes(worktree_dir: str, commands: list[str]) -> bytes:
    if not commands:
        return b""
    guard = "cd " + shlex.quote(worktree_dir) + " &&"
    return (guard + "\n" + "\n".join(commands) + "\n").encode("utf-8")
```

```python
import hashlib, os, tempfile, tomllib

def setup_hash(commands: list[str]) -> str:
    return hashlib.sha256("\n".join(commands).encode("utf-8")).hexdigest()

def load_trusted(path: str) -> dict[str, str]:
    try:
        with open(path, "rb") as fh:
            data = tomllib.load(fh)
    except (OSError, tomllib.TOMLDecodeError):
        return {}
    trusted = data.get("trusted")
    if not isinstance(trusted, dict):
        return {}
    return {k: v for k, v in trusted.items() if isinstance(v, str)}

def is_trusted(path: str, repo_id: str, commands_hash: str) -> bool:
    return load_trusted(path).get(repo_id) == commands_hash

def _esc(s: str) -> str:
    return s.replace("\\", "\\\\").replace('"', '\\"')

def _serialize_trusted(trusted: dict[str, str]) -> str:
    lines = [f'"{_esc(k)}" = "{_esc(v)}"' for k, v in trusted.items()]
    return "[trusted]\n" + "\n".join(lines) + "\n"

def record_trust(path: str, repo_id: str, commands_hash: str) -> None:
    trusted = load_trusted(path)          # read-merge (preserve prior entries)
    trusted[repo_id] = commands_hash
    text = _serialize_trusted(trusted)
    try:
        d = os.path.dirname(path) or "."
        os.makedirs(d, exist_ok=True)
        fd, tmp = tempfile.mkstemp(dir=d, prefix=".arduis-trust-")
        try:
            with os.fdopen(fd, "w", encoding="utf-8") as fh:
                fh.write(text)
            os.replace(tmp, path)
        except OSError:
            try:
                os.unlink(tmp)
            except OSError:
                pass
    except OSError:
        pass  # best-effort (a fail-closed read re-prompts anyway)
```
